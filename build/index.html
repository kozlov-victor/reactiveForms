  2. События браузера

    <form id="bind_change"> change: <input data-model="app.model" data-change="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form id="bind_click"> click: <input data-model="app.model" data-click="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form id="bind_blur"> blur: <input data-model="app.model" data-blur="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form id="bind_focus"> focus: <input data-model="app.model" data-focus="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form data-submit="modelToResult()" id="bind_submit"> submit: <input data-model="app.model"> <div> {{app.result}} </div> <input type="submit" value="submit"> </form> <hr/> <form id="bind_keypress"> keypress: <input data-model="app.model" data-keypress="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form id="bind_keyup"> keyup: <input data-model="app.model" data-keyup="modelToResult()"> <div> {{app.result}} </div> </form> <hr/> <form id="bind_keydown"> keydown: <input data-model="app.model" data-keydown="modelToResult()"> <div> {{app.result}} </div> </form> <hr/>
    var evnts = [ 'change', 'click', 'blur', 'focus', 'submit', 'keypress', 'keyup', 'keydown' ]; for (var i=0;i<evnts.length;i++) { var e = evnts[i]; RF.applyBindings('#bind_'+e,{ app: { model: 'appModel', result:'appResult' }, modelToResult: function(){ this.app.result = this.app.model; } }); }

  14. Компонент, который взаимодействует с моделью из внешнего источника данных

    <style> .counter { display: inline-block; border: 1px solid black; } </style> <div id="counterTmpl" style="display:none;"> <div class="counter"> <input type="number" data-model="model.user[modelField]"> <input type="range" min="1" max="200" data-model="model.user[modelField]"> <button data-click="inc()">+</button> <button data-click="dec()">-</button> </div> </div> <div id="application"> <div> дебаг модели: {{model}} </div> <div> name: <input data-model="model.user.name"> </div> <div> current power: {{model.user.power}} </div> <div> current carma level: {{model.user.karmaLevel}} </div> <div> power: <counter-component data-properties=" { modelField: 'power' }"> </counter-component> </div> <div> karma level: <counter-component data-properties=" { modelField: 'karmaLevel' }"> </counter-component> </div> </div>
    var applicationModel = { user: { name: 'defaultUserName', power: 100, karmaLevel: 42 } }; RF.registerComponent('counter-component',{ template: { type: 'dom', value: 'counterTmpl' }, external: { // model объявлена как external, это значит, что она ссылается на внешний объект // если модель не будет обозначена как external, в компоненте будет создана локальная копия модели model: applicationModel }, modelField: '', // это свойство индивидуально для каждого компонента и будет определено (переопределено) в data-properties inc: function(){ this.model.user[this.modelField]++; }, dec: function(){ this.model.user[this.modelField]--; } }); RF.applyBindings('#application',{ model: applicationModel });

  15. Циклы

    <style> .pointer { cursor: pointer; } .selected { outline: 1px solid darkblue; } </style> <div id="app"> <div> Текущий фрукт: {{currentFruit.name}} </div> <select data-model="currentFruit"> <option data-value="item" data-for="item in fruits"> {{item.name}} </option> </select> <table> <tr class="pointer" data-class="{selected:item==currentFruit}" title="применить" data-click="setCurrentFruit(item)" data-for="item,index in fruits"> <td>{{item.name}}</td> <td><button data-click="remove(index)">-</button></td> </tr> </table> <button data-click="add()">add</button> </div>
    var fruits = [{name:'apple'},{name:'peanut'},{name:'orange'}]; RF.applyBindings('#app',{ fruits: fruits, currentFruit: fruits[0], add: function(){ this.fruits.push({name:~~(Math.random()*100)}); }, remove: function(i){ this.fruits.splice(i,1); }, setCurrentFruit: function(currentFr){ this.currentFruit = currentFr; } });

  16. Вложенные циклы

    <div id="app"> <ul> <li data-for="b,i in app.bundles"> {{i+1}}. {{b.name}} <button data-click="app.removeExternal(i)">remove({{i}})</button> <ul> <li data-click="r()" data-for="d,j in b.data"> {{j+1}}. {{d.slot}} <button data-click="app.removeInternal(i,j)">remove({{i}},{{j}})</button> </li> </ul> <button data-click="app.addInternal(i)">+</button> </li> </ul> <button data-click="app.add()">+</button> <hr/> <button data-click="app.changeExternal()">Изменить внешнее значение</button> <button data-click="app.changeInternal()">Изменить вложенное значение</button> <button data-click="app.changeAll()">Изменить все (изменить по ссылке)</button> <button data-click="app.changeGlobal()">Изменить ссылку на объект</button> </div>
    var app = {}; app.bundles = [ { name:'Хранилище 1', data: [ {'slot':12}, {'slot':33} ] }, { name:'Хранилище 2', data: [ {'slot':4}, {'slot':32} ] }, { name:'Хранилище 3', data: [ {'slot':2}, {'slot':45} ] } ]; var rnd = function(){ return Math.random().toFixed(3) }; app.changeExternal = function(){ for (var i=0;i<this.app.bundles.length;i++) { this.app.bundles[i].name = 'bundle_'+rnd(); } }; app.changeInternal = function(){ for (var i=0;i<this.app.bundles.length;i++) { for (var j=0;j<this.app.bundles[i].data.length;j++) { this.app.bundles[i].data[j] = {slot:'slot_'+rnd()}; } } }; app.changeAll = function(){ for (var i=0;i<this.app.bundles.length;i++) { this.app.bundles[i].name = 'bundle_'+rnd(); for (var j=0;j<this.app.bundles[i].data.length;j++) { this.app.bundles[i].data[j] = {slot:'slot_'+rnd()}; } } }; app.add = function(){ this.app.bundles.push({ name:'Хранилище_'+rnd(), data: [ {'slot':'sl_'+rnd()}, {'slot':'sl_'+rnd()} ] }); }; app.removeExternal = function(i){ this.app.bundles.splice(i,1); }; app.removeInternal = function(i,j){ this.app.bundles[i].data.splice(j,1); }; app.addInternal = function(i){ this.app.bundles[i].data.push({ slot: 'new_'+rnd() }); }; app.addInternal = function(i){ this.app.bundles[i].data.push({slot:'new_'+rnd()}) }; app.changeGlobal = function(){ this.app.bundles = [ { name:'Хранилище 1'+rnd(), data: [ {'slot':12+rnd()}, {'slot':33+rnd()} ] }, { name:'Хранилище 2'+rnd(), data: [ {'slot':4+rnd()}, {'slot':32+rnd()} ] }, { name:'Хранилище 3'+rnd(), data: [ {'slot':2+rnd()}, {'slot':45+rnd()} ] } ]; }; RF.applyBindings('#app',{ app:app, r:function(){ } });

  17. Рекурсивные шаблоны (древовидная структура)

    <div id="nodeItemTmpl"> <span data-click="triggerCollapse()"> {{collapsed?'+':'-'}} {{node.name}} </span> <ul data-if="!collapsed"> <li data-for="item in node.children"> <node-item data-properties="{node:item}"></node-item> </li> </ul> </div> <div id="app"> <ul> <li data-for="item in app.nodes"> <node-item data-properties="{node:item}"></node-item> </li> </ul> </div>
    var app = {}; app.nodes = [ { name:'node_1', children: [ { name:'node_1_1' }, { name:'node_1_2' } ] }, { name:'node_2', children: [ { name:'node_2_1' }, { name:'node_2_2', children: [ {name:'node_2_2_1'}, { name:'node_2_2_2', children: [ {name:'node_2_2_2_1'}, { name:'node_2_2_2_2' } ] } ] } ] } ]; RF.registerComponent('node-item',{ template: { type:'dom', value:'nodeItemTmpl' }, collapsed: true, triggerCollapse: function(){ this.collapsed = !this.collapsed; }, external: { node:null } }); RF.applyBindings('#app',{ app:app });

    <div style="display: none;"> <div id="page1Tmpl"> Введите имя <input data-model="app.user.firstName"> <div> <button data-click="app.navigateTo(2)">Далее</button> </div> </div> <div id="page2Tmpl"> {{app.user.firstName}}, Введите фамилию <br/><input data-model="app.user.lastName"> <div> уровень кармы <br/><input data-model="app.user.carmaLevel"> </div> <div> <button data-click="app.goBack()">Назад</button> <button data-click="app.navigateTo(3)">Далее</button> </div> </div> <div id="page3Tmpl"> Привет, {{app.user.firstName}} {{app.user.lastName}} <br/> Уровень кармы {{app.user.carmaLevel}} <div> <button data-click="app.goBack()">Назад</button> </div> </div> </div> <div data-route=""></div>
    var app = { user: { firstName:'', lastName:'', carmaLevel:42 }, navigateTo: function(index){ RF.Router.navigateTo('page'+index); }, goBack: function(){ RF.Router.goBack(); } }; var page1 = RF.registerComponent('page1',{ template: { type: 'dom', value: 'page1Tmpl' }, external: { app: app } }); var page2 = RF.registerComponent('page2',{ template: { type: 'dom', value: 'page2Tmpl' }, external: { app: app } }); var page3 = RF.registerComponent('page3',{ template: { type: 'dom', value: 'page3Tmpl' }, external: { app: app } }); RF.Router.setup({ 'page1': page1, 'page2': page2, 'page3': page3 }); RF.Router.navigateTo('page1');
